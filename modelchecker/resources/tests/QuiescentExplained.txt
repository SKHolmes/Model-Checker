processes {
 //Quiescent traces can ALL have x?* appended for any x
A = s!->a!->STOP|s!->b?->STOP.          //Qt  (a!),(b?)
B = s!->a!->STOP.     
          
Ab = s!->a!->b?->STOP|s!->b?->STOP.     //Qt  (a!),(b?)
Abc = s!->a!->b?->c?->STOP|s!->b?->STOP. //Qt  (a!), (b?), 

S = a!->c!->STOP.
Sb = a!->b?->c!->STOP.
S2b = a!->b?->b?->c!->STOP.
S3b = a!->b?->b?->b?->c!->STOP.

R = a!->STOP.
Rb = a!->b?->STOP.
R2b = a!->b?->b?->STOP.
R3b = a!->b?->b?->b?->STOP.
}


operation {
Sb !<q S.
S !<q Sb.
Sb !<q S2b.
S2b !<q Sb.
S3b !<q S2b.
S2b !<q S3b.

Rb <q R.
R <q Rb.
Rb <q R2b.
R2b <q Rb.
R3b <q R2b.
R2b <q R3b.

Rb !<q{cong} R.
R2b !<q{cong} Rb.
R3b !<q{cong} R2b.

R !<q{cong} Rb.
Rb !<q{cong} R2b.
R2b !<q{cong} R3b.

A <q  B. 
B !<q A.

Ab <q  B. 
B !<q Ab.

Ab <q A.
A <q Ab.

Abc <q Ab.

Ab <q Abc.

}
