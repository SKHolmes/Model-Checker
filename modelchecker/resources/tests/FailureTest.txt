processes {
A = a->b->STOP|a->c->STOP|a->(b->STOP|c->STOP).
B = a->b->STOP|a->c->STOP.
C = a->(b->STOP|c->STOP).

Al = a->b->Al|a->c->Al|a->(b->Al|c->Al).
Bl = a->b->Bl|a->c->Bl.

Aa = a->s->b->STOP|a->s->c->STOP|a->s->(b->STOP|c->STOP).
Ba = s->a->b->STOP|a->s->c->STOP.

Bs = a->s->b->STOP|a->s->c->STOP.


L= a->a->X|a->b->STOP|a->(a->Y|b->STOP), 
    Y= a->Y,
    X=a->X.
Ll = a->a->a->a->X|a->b->STOP,
    X=a->X.
}
automata A,B,C,Al,Bl,Aa,Ba,Bs,L,Ll.

operation {
A!~B.

A*B.

A!*C.
Al!~Bl.
Al*Bl.

Aa!*Ba.
Aa!~Bs.
Aa*Bs.

L!~Ll.
L*Ll.
}

processes {
X= a->b->a->a->X.
Y = c->c->c->Y.
Z= (X||Y)/{a/c}.

Q = Z\{a}.
W = simp(abs(Q)).
}

automata Z,W,Q.


processes {
X1= a->b->X1|a->a->X1.
Y1 = c->c->Y1|c->c->Y1.
Z1 = (X1||Y1)/{a/c}.

Q1 = Z1\{a}.
W1 = simp(abs(Q1)).

Goal = b->Goal.
}

automata Z1,W1,Q1,Goal.

operation {
Goal~ W1.
Goal ~ W.
}
