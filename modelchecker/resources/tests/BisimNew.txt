const N = 2.
processes {
A = a->B,
    B = (b->A | c->A).
Ap = A || A.

B = a -> C,
    C = (b->B | c->ERROR).
Bp = B || B.
ABp = A || B.
D = aaa -> C,
    C = (bb->STOP | cc->STOP).
Dp = D || D.
X = red -> Y,
   Y = (red ->Y|red ->X).
Z = simp(X).


Q = a->a->a->b->STOP.
P = a->a->b->STOP.
}

automata A,Ap.
automata B,Bp.
automata ABp.
automata D,Dp.
automata X,Z,Q,P.

operation {
Ap~ A.
Bp~B.
Dp ~ D.
ABp ~ B.

P!~Q.
X ~ Z.
}
