  processes {
        A = (a -> b? -> c -> STOP || x -> b! -> z -> STOP).

        ExpectedA = (x -> B | a -> x -> C),
            B = (a -> C | b! -> (a -> z -> STOP | z -> a -> STOP)),
            C = (b! -> (c -> z -> STOP | z -> c -> STOP)).

        B = (B||C),
            B = (a?->b->c!->STOP),
            C = (a!->x->c?->STOP).

        ExpectedB = (a! -> (b -> B | x -> b -> C)),
            B = (x -> C | c! -> x -> STOP),
            C = (c! -> STOP).

               C = (a -> b? -> c -> STOP || x -> b? -> z -> STOP).

        ExpectedC = (a -> B | x -> C),
            B = (x -> D | b? -> (c-> (x ->(b?->F|c->Nw))| x->Nw)),
            New = b?->E,
            Nw = b?->E,
            C = (a -> D | b? -> (z-> (a->(b?->G|z->New))| a->New)),
            D = (b? -> E),
            E = (c -> F | z -> G),
            F = (z -> H),
            G = (c -> H),
            H = (STOP).

    }

    automata A,ExpectedA.

    automata B,ExpectedB.

    automata C,ExpectedC.

    operation {
        A ~ ExpectedA.
        B ~ ExpectedB.
        C ~ ExpectedC.
    }
